import requests
words=["dog","cat"]
for word in words:
    r=requests.get("http://mall.baicizhan.com/ws/search?w={}".format(word))
    print(r.text)
    print(r.json()["img"])
    print(u"http://baicizhan.qiniucdn.com/word_audios/{}.mp3".format(word))
